@Component
@Configuration
class MQService {

    val LOGGER = getLogger(javaClass)

    @Autowired
    lateinit var mqConnection: MQConnection

    @Value("\${ibmmq.timeout}")
    private var timeout: Long = 0

    val healthStatus: String
        get() {

            val response: String
            val mqConFactory = mqConnection.getMQConnectionFactory(true)

            if (mqConFactory != null) {
                response = "Sucessfull.Avaloq Connection is up"
            } else {
                response = "Unsucessfull.Avaloq Connection is down"
            }
            return response
        }


    @Throws(MQTimeOutException::class)
    fun sendAndRecieve(message: String, msguid: String, jmsTemplateContext: JMSTemplateContext, country: String): String {
        return callMQ(message, msguid, jmsTemplateContext,country)
    }

    @Throws(MQTimeOutException::class)
    private fun callMQ(message: String, msguid: String, jmsTemplateContext: JMSTemplateContext, country: String): String {
        var responseMessage: String = ""
        var senderQueue : MQQueue = MQQueue(jmsTemplateContext.requestDestination)
        var receiverQueue: MQQueue = MQQueue(jmsTemplateContext.responseDestination)
        try {
            LOGGER.info("Sending Avaloq Request : Request No {} : Payload {}, country {}", msguid, message,country)
          if(country.equals("hk")) {
              senderQueue = MQQueue(jmsTemplateContext.requestDestinationhk)
              receiverQueue = MQQueue(jmsTemplateContext.responseDestinationhk)
          }
            senderQueue.targetClient = 1
            val processor = CorrelationIdPostProcessor(msguid, senderQueue, receiverQueue)
            val jmsTemplate = JmsTemplate(mqConnection.getMQConnectionFactory(true))

            LOGGER.info("jmsTemplate ->{}", jmsTemplate.hashCode())

            jmsTemplate.isExplicitQosEnabled = true
            jmsTemplate.setReceiveTimeout(timeout)
            jmsTemplate.defaultDestination = senderQueue
            jmsTemplate.convertAndSend(senderQueue, message, processor)
            val messageId = processor.message.jmsMessageID
            val startTime = System.currentTimeMillis()
            responseMessage = jmsTemplate.receiveSelectedAndConvert(receiverQueue, "JMSCorrelationID = '$messageId'").toString()
            val endTime = System.currentTimeMillis()
            val elapsedTime = endTime - startTime
            LOGGER.info("callMQ Elapsed time for receiving the response:{}", elapsedTime)
            LOGGER.info("callMQ Timeout value set is:{}", timeout)
            if (elapsedTime > timeout) {
                LOGGER.info("Failed to receive AVALOQ response in outbound que with specified time {} : Request No {} : Response {}", elapsedTime, msguid, responseMessage)
                throw MQTimeOutException("AVALOQ failed to respond within stipulated time frame")
            } else {
                LOGGER.info("Fetched Avaloq Response for : Request No {} : Response {}", messageId, responseMessage)
            }
            return responseMessage
        } catch (ex: Exception) {
            println(ex.localizedMessage)
            println(ex)
            LOGGER.error("Failed to receive AVALOQ response in outbound que with specified time  for request no message {}", message)
            throw MQTimeOutException("AVALOQ failed to respond within stipulated time frame")
        }

        return responseMessage
    }

    companion object {
        private val connectionFactory: MQQueueConnectionFactory? = null
    }
}

private class CorrelationIdPostProcessor(private val correlationId: String, private val senderQueue: MQQueue, private val receiverQueue: MQQueue) : MessagePostProcessor {
    private val messageRef = AtomicReference<Message>()

    @Throws(JMSException::class)
    override fun postProcessMessage(msg: Message): Message {
        msg.jmsCorrelationID = correlationId
        msg.jmsReplyTo = receiverQueue
        msg.jmsDestination = senderQueue
        messageRef.set(msg)
        return msg
    }

    val message: Message
        get() = messageRef.get()

}

@Configuration
class MQConnection {

    // New CCDT property
    @Value("\${ibmmq.ccdtUrl}")
    private val ccdtUrl: Resource? = null

    @Value("\${ibmmq.queuemanager}")
    private val queueManager: String? = null

    @Value("\${ibmmq.keystore}")
    private val keyStore: Resource? = null

    @Value("\${ibmmq.truststore}")
    private val trustStore: Resource? = null

    @Value("\${ibmmq.keypassword}")
    private val keyPassword: String? = null

    @Value("\${ibmmq.trustpassword}")
    private val trustPassword: String? = null

    @Bean
    fun getMQConnectionFactory(retry: Boolean): MQQueueConnectionFactory {
        if (connectionFactory != null && !retry) {
            LOGGER.info("Successfully returned cached MQ connection factory")
            return connectionFactory!!
        }
        createConnectionFactory()
        return connectionFactory!!
    }

    private fun createConnectionFactory() {
        // Required for non-IBM JREs to map TLS ciphers correctly
        System.setProperty("com.ibm.mq.cfg.useIBMCipherMappings", "false")
        try {
            val sslContext = SSLContexts.custom()
                .loadTrustMaterial(trustStore!!.url, trustPassword!!.toCharArray())
                .loadKeyMaterial(keyStore!!.url, keyPassword!!.toCharArray(), keyPassword.toCharArray())
                .build()

            val socketFactory = sslContext.socketFactory

            val factory = MQQueueConnectionFactory()

            // 1. Set CCDT URL (Resolves classpath: to a URL MQ understands)
            ccdtUrl?.let {
                factory.setCCDTURL(it.url)
                LOGGER.info("CCDT URL Request $it.url")
            }

            // 2. Set Queue Manager (use "*" to let CCDT decide or specific name)
            factory.queueManager = queueManager ?: "*"

            LOGGER.info("MQ queueManager $queueManager")
            // 3. Set SSL Socket Factory (MQ will use this for the handshake)
            factory.sslSocketFactory = socketFactory

            // 4. Client Transport is mandatory for CCDT
            factory.transportType = WMQConstants.WMQ_CM_CLIENT

            factory.sslFipsRequired = false

            connectionFactory = factory
            LOGGER.info("MQ Connection Factory Created using CCDT")

        } catch (ex: Exception) {
            // Caught as generic Exception for brevity, but matches your specific list
            LOGGER.error("Exception in getting Avaloq connection, queue sender: ", ex)
        }
    }

    companion object {
        private val LOGGER = getLogger(MQConnection::class.java)
        private var connectionFactory: MQQueueConnectionFactory? = null
    }
}
